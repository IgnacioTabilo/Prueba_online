import ipywidgets
import numpy as np
import matplotlib.pyplot as plt

X = np.random.uniform(0, 5, size=100)
noise = np.random.normal(size=100)
def plot_fct(w=1):
    y = 2 * X + w * noise
    plt.scatter(X, y)

ipywidgets.interact(plot_fct, w=(0, 5, 0.5))
